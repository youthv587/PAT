#include<iostream>
#include<algorithm>
#include<string>
using namespace std;

    int n;
while(cin>>n){
string str[n];
for(int i=0;i<n;i++)
        cin>>str[i];
sort(str,str+n);
              
for(int i=0;i<n;i++)
        cout<<str[i]<<" ";              
}

    return 0;
}
